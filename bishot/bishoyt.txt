#include <iostream>
#include <string>
using namespace std;
struct Time {
    int hours;
    int minutes;
    int seconds;
};
struct movie {
    string name;
    Time duration;
    float price;
    int year;
    string language;
};
movie inputmovie()
{
    movie m;
    cout << "Enter movie name : ";
    cin >> m.name;
    cout << "Enter movie in hours : ";
    cin >> m.duration.hours;
    cout << "Enter movie in minutes :";
    cin >> m.duration.minutes;
    cout << "Enter movie in seconds :";
    cin >> m.duration.seconds;
    cout << "Enter movie price : ";
    cin >> m.price;
    cout << "Enter movie year : ";
    cin >> m.year;
    cout << "Enter movie language : ";
    cin >> m.language;

    return m;
}
struct MovieRental {
    string title;
    string customer;
    int rentalPeriod;
    float rentalFee;
};
const int MAX_MOVIES = 100;
MovieRental rentedMovies[MAX_MOVIES];

int main() {
    inputmovie();
    int numMoviesRented = 0;
  
        cout << "Enter the movie title: ";
        getline(cin, rentedMovies[0].title);

        cout << "Enter the customer name: ";
        getline(cin, rentedMovies[0].customer);

        cout << "Enter the rental period (in days): ";
        cin >> rentedMovies[0].rentalPeriod;

        cout << "Enter the rental fee: ";
        cin >> rentedMovies[0].rentalFee;

        numMoviesRented++;
    

    for (int i = 0; i < numMoviesRented; i++) {
        cout << "Movie #" << i + 1 << endl;
        cout << "Title: " << rentedMovies[i].title << endl;
        cout << "Customer: " << rentedMovies[i].customer << endl;
        cout << "Rental period: " << rentedMovies[i].rentalPeriod << " days" << endl;
        cout << "Rental fee: $" << rentedMovies[i].rentalFee << endl;
        cout << endl;
    }

}
